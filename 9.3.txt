#include<reg51.h>
#include"1602lcd.h"	  
#define uchar unsigned char	
#define uint unsigned int
uchar ge,shi,bai,f,boom_on;
uint count,a;
sbit boom=P3^6;
void display();
void fenjie();
void main()
{
	lcd_init();
	lcd_pos(1,1);
	lcdwrite_zm('C');
	lcdwrite_zm('O');
	lcdwrite_zm('U');
	lcdwrite_zm('N');
	lcdwrite_zm('T');
	lcdwrite_zm(':');

	TMOD=0x15;
	TH0=0;
	TL0=0;
	TH1=(65536-5000)/256;
	TL1=(65536-5000)%256;
	ET1=1;
	EA=1;
	TR0=1;
	while(1)
	{
		count=TH0*256+TL0;
		fenjie();
		display();
		if(count%10==0&&count!=0)
		{
			if(boom_on!=1)
			{
				boom_on=1;
				TR1=1;
			}
		}
		else if(f==1)
		{
			boom_on=0; 
			f=0;
		}	
	}
}
void fenjie()
{
	bai=count/100;
	shi=count%100/10;
	ge=count%10;	
}
void display()
{
	lcd_pos(1,8);
	lcdwrite_sz(bai);
	lcdwrite_sz(shi);
	lcdwrite_sz(ge);	
}
void timero() interrupt 3
{
	TH1=(65536-5000)/256;
	TL1=(65536-5000)%256;
	boom=~boom;
	a++;
	if(a==200)
	{
		f=1;
		a=0;
		TR1=0;
	}
	if(f!=1)
	{
		TR1=1;
	}	
}
